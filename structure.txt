.
├── data/              # Всі дані проекту (сирі, проміжні, оброблені)
│   ├── raw/           # Оригінальні, незмінні сирі дані
│   ├── interim/       # Проміжні дані, отримані в процесі обробки
│   └── processed/     # Очищені та фінально оброблені дані, готові для використання (наприклад, бекендом)
|
├── backend/           # Код бекенд-сервісів (наприклад, API на Python/FastAPI, Node.js/Express тощо)
│   ├── src/           # Основний вихідний код бекенду
│   ├── tests/         # Тести для бекенду
│   ├── config/        # Файли конфігурації бекенду
│   ├── Dockerfile     # (Опціонально) Файл для збірки Docker-образу бекенду
│   └── requirements.txt # Залежності бекенду (для Python) або package.json (для Node.js)
|
├── frontend/          # Код фронтенд-додатку (наприклад, React)
│   ├── public/        # Статичні файли (index.html тощо)
│   ├── src/           # Основний вихідний код React-додатку
│   ├── public/
│   ├── package.json   # Залежності фронтенду (npm/yarn)
│   ├── Dockerfile     # (Опціонально) Файл для збірки Docker-образу фронтенду
│   └── ... (інші стандартні файли React-проекту)
|
├── modeling/          # Скрипти, ноутбуки та код, пов'язаний з тренуванням та оцінкою моделей
│   ├── notebooks/     # Ноутбуки для експериментів та досліджень
│   ├── scripts/       # Скрипти для автоматизації тренування, оцінки
│   └── src/           # Перевикористовуваний код моделей (наприклад, класифікатори, трансформери)
|
├── packages/          # (Опціонально) Перевикористовуваний код або бібліотеки, що використовуються кількома компонентами (наприклад, бекендом та скриптами моделювання)
│   ├── common_utils/  # Загальні утиліти
│   └── shared_models/ # Код для завантаження/використання тренованих моделей у бекенді
|
├── artifacts/         # Результати роботи моделювання: треновані моделі, scaler-и, звіти, графіки
|
├── scripts/           # Допоміжні скрипти для автоматизації завдань (збірка, розгортання, міграції даних)
|
├── docs/              # Додаткова документація проекту (діаграми, API-специфікації, пояснення)
|
├── docker-compose.yml # (Опціонально) Для запуску всіх сервісів (бекенд, фронтенд, база даних) разом
├── .gitignore         # Файл для Git, що вказує, які файли ігнорувати
└── README.md          # Основний опис проекту, інструкції зі встановлення та запуску
